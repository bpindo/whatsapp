<script>
let sa = "", sp = "", auto = false, loading = false, totalScanned = 0;
const scannedSet = new Set();
let historyData = JSON.parse(localStorage.getItem("hive_waiv_ultra_store")) || [];

function switchTab(i) {
    document.querySelectorAll(".tab").forEach((t, idx) => t.classList.toggle("active", idx === i));
    document.querySelectorAll(".page").forEach((p, idx) => p.classList.toggle("active", idx === i));
}

async function processBatch() {
    if (!auto || loading) return;
    loading = true;
    try {
        const query = { limit: 20, tag: "waiv" };
        if (sa) { query.start_author = sa; query.start_permlink = sp; }
        const r = await fetch("https://api.hive.blog", {
            method: "POST",
            body: JSON.stringify({ jsonrpc: "2.0", method: "condenser_api.get_discussions_by_created", params: [query], id: 1 })
        });
        const j = await r.json();
        const posts = j.result || [];
        totalScanned += posts.length;
        document.getElementById("counter").textContent = "Scan: " + totalScanned;
        posts.forEach(x => {
            const url = `https://hive.blog/@${x.author}/${x.permlink}`;
            if (!scannedSet.has(url)) {
                scannedSet.add(url);
                const payoutValue = parseFloat(x.pending_payout_value.split(" ")[0]);
                if (payoutValue >= 0.07 && payoutValue <= 1.50) {
                    const created = new Date(x.created + 'Z');
                    const diffDays = ((new Date() - created) / (1000 * 60 * 60 * 24)).toFixed(2);
                    if (!historyData.some(item => item.url === url)) {
                        historyData.unshift({ url, days: diffDays, payout: x.pending_payout_value });
                    }
                }
            }
        });
        if (posts.length) {
            sa = posts[posts.length - 1].author;
            sp = posts[posts.length - 1].permlink;
            localStorage.setItem("hive_waiv_ultra_store", JSON.stringify(historyData));
            updateHistoryUI();
        }
    } catch (e) {}
    loading = false;
    document.getElementById("progress").textContent = `Terjaring: ${historyData.length}`;
    if (auto) setTimeout(processBatch, 1500);
}

function updateHistoryUI() {
    const container = document.getElementById("historyContainer");
    const tabLabel = document.getElementById("historyTab");
    const emptyMsg = document.getElementById("historyEmpty");
    container.innerHTML = "";
    tabLabel.textContent = `📜 (${historyData.length})`;
    emptyMsg.style.display = historyData.length ? "none" : "block";

    historyData.forEach((item) => {
        const div = document.createElement("div");
        div.className = "item";
        div.innerHTML = `
            <div class="url-wrapper">
                <a href="${item.url}" target="_blank" class="main-link">${item.url}</a>
            </div>
            <div class="badge-row">
                <span class="badge badge-age">${item.days}d</span>
                <span class="badge badge-payout">${item.payout.split(" ")[0]}</span>
            </div>`;
        container.appendChild(div);
    });
}

function clearHistory() {
    if (confirm("Hapus semua?")) {
        historyData = [];
        localStorage.setItem("hive_waiv_ultra_store", JSON.stringify(historyData));
        updateHistoryUI();
    }
}

function copyAllHistory() {
    if (historyData.length === 0) return;
    navigator.clipboard.writeText(historyData.map(i => i.url).join("\n")).then(() => alert("Kopi!"));
}

function toggleAuto() {
    auto = !auto;
    const btn = document.getElementById("autoBtn");
    btn.textContent = auto ? "⏹ STOP" : "▶ START HUNTER";
    btn.className = "btn-main " + (auto ? "btn-stop" : "btn-start");
    if (auto) processBatch();
}

window.onload = updateHistoryUI;
</script>

<script>
const apiUrl = "https://script.google.com/macros/s/AKfycbzzPdLkT1PPda_r_RdTLYIWbiIIQ7hWvUc8-_s3FeF6xXV4rEX9mCQ_40xUU9YH6jCE/exec";
let voters = [], disables = [], link = "";

// Toast helper
function toast(msg){
  const t = document.getElementById("toast");
  t.innerText = msg;
  t.classList.add("show");
  setTimeout(()=>t.classList.remove("show"),2500);
}

// Tabs
function showTab(tabId){
  document.querySelectorAll('.tab-content').forEach(c=>c.classList.remove('active'));
  document.getElementById(tabId).classList.add('active');
  document.querySelectorAll('.tab').forEach(t=>t.classList.remove('active'));
  event.target.classList.add('active');
}

// Paste JSON ke textarea
async function pasteToTextarea(id){
  try{
    const text = await navigator.clipboard.readText();
    document.getElementById(id).value = text.trim();
    toast("JSON berhasil dipaste");
  }catch{
    toast("Clipboard ditolak");
  }
}

// Load Data
function loadData(){
  fetch(apiUrl+"?api=list")
  .then(res=>res.json())
  .then(data=>{
    voters = data.voters || [];
    disables = data.disable || [];
    link = data.link || "";
    refreshVoter();
    refreshDisable();
    document.getElementById("currentLink").innerText = link;
    document.getElementById("jsonVoters").value = JSON.stringify(voters,null,2);
    document.getElementById("jsonDisable").value = JSON.stringify(disables,null,2);
    document.getElementById("status").innerText = "Data siap.";
  });
}

// Voter Functions
function refreshVoter(){
  const sel = document.getElementById("voterDropdown");
  sel.innerHTML = "";
  voters.forEach(v=>{
    const opt = document.createElement("option");
    opt.value = v.username;
    opt.innerText = v.username + " (" + (v.aktif ? "Aktif" : "Nonaktif") + ")";
    sel.appendChild(opt);
  });
}

function addVoter(){
  const username = document.getElementById("newUsername").value.trim().toLowerCase();
  const postingKey = document.getElementById("newPostingKey").value.trim();
  const aktif = document.getElementById("newActive").checked;
  if(!username || !postingKey){ toast("Isi username & posting key"); return; }
  fetch(apiUrl,{method:'POST',body: JSON.stringify({api:'addVoter', username, postingKey, aktif})})
  .then(res=>res.json()).then(res=>{
    toast(res.message);
    loadData();
    document.getElementById("newUsername").value="";
    document.getElementById("newPostingKey").value="";
  });
}

function deleteVoter(){
  const username = document.getElementById("voterDropdown").value;
  if(!username) return toast("Pilih voter");
  fetch(apiUrl,{method:'POST',body: JSON.stringify({api:'deleteVoter', username})})
  .then(res=>res.json()).then(res=>{
    toast(res.message);
    loadData();
  });
}

function toggleVoter(){
  const username = document.getElementById("voterDropdown").value;
  if(!username) return toast("Pilih voter");
  fetch(apiUrl,{method:'POST',body: JSON.stringify({api:'toggleVoter', username})})
  .then(res=>res.json()).then(res=>{
    toast(res.message);
    loadData();
  });
}

function updateVoters(){
  let arr;
  try{ arr = JSON.parse(document.getElementById("jsonVoters").value); if(!Array.isArray(arr)) throw "Bukan array"; }
  catch(e){ toast("Format JSON salah!"); return; }
  fetch(apiUrl,{method:'POST',body: JSON.stringify({api:'updateVoters', voters: arr})})
  .then(res=>res.json()).then(res=>{
    toast(res.message);
    loadData();
  });
}

// Disable Functions
function refreshDisable(){

  sel.innerHTML = "";
  disables.forEach(u=>{
    const opt = document.createElement("option");
    opt.value = u.username;
    opt.innerText = u.username;
    sel.appendChild(opt);
  });
}

function deleteDisable(){
  const username = document.getElementById("disableDropdown").value;

  .then(res=>res.json()).then(res=>{
    toast(res.message);
    loadData();
  });
}

function updateDisable(){

    loadData();
  });

// Move Functions
function moveDisableToVoter(){
  const username = document.getElementById("disableDropdown").value;
  if(!username) return toast("Pilih username");

  let disableObj = disables.find(d => d.username === username);
  if(!disableObj) return toast("Data disable tidak ditemukan");

  fetch(apiUrl, {method: 'POST', body: JSON.stringify({api: 'addVoter', username: disableObj.username, postingKey: disableObj.postingKey})})
  .then(res => res.json())
  .then(res => {
    toast(res.message);
    if(res.status === "success"){
      fetch(apiUrl, {method: 'POST', body: JSON.stringify({api: 'deleteDisable', username: disableObj.username})})
      .then(()=>loadData());
    }
  });
}

function moveVoterToDisable(){
  const username = document.getElementById("voterDropdown").value;
  if(!username) return toast("Pilih voter");

  let voterObj = voters.find(v => v.username === username);
  if(!voterObj) return toast("Voter tidak ditemukan");

  fetch(apiUrl, {method: 'POST', body: JSON.stringify({api: 'addDisable', username: voterObj.username, postingKey: voterObj.postingKey})})
  .then(res => res.json())
  .then(res => {
    toast(res.message);
    if(res.status === "success"){
      fetch(apiUrl, {method: 'POST', body: JSON.stringify({api: 'deleteVoter', username: voterObj.username})})
      .then(()=>loadData());
    }
  });
}

    toast("Link diperbarui");
    loadData();
    document.getElementById("newLink").value="";
  });
}

async function cekVP(username){
    try {
        let res = await fetch("https://api.hive.blog", {
            method: "POST",
            body: JSON.stringify({
                jsonrpc: "2.0", method: "condenser_api.get_accounts",
                params: [[username]], id: 1
            })
        });
        let j = await res.json();
        if(!j.result || j.result.length === 0) return null;
        
        let acc = j.result[0];
        let vp = acc.voting_power;
        let last = new Date(acc.last_vote_time + "Z");
        let now = new Date();
        let seconds = (now - last) / 1000;
        let regen = seconds * 10000 / 432000;
        let current = Math.min(10000, vp + regen);
        return current / 100;
    } catch(e) {
        return null;
    }
}

// Fungsi pembungkus untuk memproses tiap akun dalam batch
async function processAccount(acc) {
    let vp = await cekVP(acc.username);
    if (vp === null) {
        log(acc.username + ": Gagal cek data", "err");
        return;
    }
    
    let isAktif = vp >= 96;
    let statusLabel = isAktif ? "VOTER" : "DISABLED";
    let statusClass = isAktif ? "ok" : "warn";
    
    log(`${acc.username} [${vp.toFixed(2)}%] → ${statusLabel}`, statusClass);
    
    // Update data lokal
    acc.aktif = isAktif;
}

    for (let i = 0; i < voters.length; i += BATCH_SIZE) {

        log(`Memproses batch ${Math.floor(i/BATCH_SIZE) + 1}...`, "batch-info");

        await Promise.all(currentBatch.map(acc => processAccount(acc)));

function apiAddVoter(username, postingKey) {

  if (voters.some(v => v.username === username)) return { status: "error", message: "Username sudah ada" };
  voters.push({ username, postingKey });
  saveHiveVoters(voters);

  let voters = getHiveVoters();
  let i = voters.findIndex(v => v.username === oldUsername);
  if (i === -1) return { status: "error", message: "Username tidak ditemukan" };
  voters[i] = { username: newUsername, postingKey: newPostingKey };

<script>
const voters = ${JSON.stringify(voters)};
const candidate = "${candidate}";

function log(msg){
  const box=document.getElementById("log");
  box.innerHTML+=msg+"<br>";
  box.scrollTop=box.scrollHeight;
}

function startVote(){
  if(!candidate){
    log("❌ Candidate kosong");
    return;
  }
  if(!voters.length){
    log("❌ Tidak ada voter aktif");
    return;
  }

  const parts = candidate.split("/");
  const author = parts[4].replace("@","");
  const permlink = parts[5];

  log("Candidate: "+candidate);
  log("Author: "+author);
  log("Permlink: "+permlink);

  voteLoop(author, permlink, 0);
}

function voteLoop(author, permlink, i){
  if(i >= voters.length){
    log("✅ Selesai semua vote");
    document.getElementById("status").innerText = "Vote selesai";
    return;
  }

  const voter = voters[i];
  log("Voting: "+voter.username);

  hive.broadcast.vote(
    voter.postingKey,
    voter.username,
    author,
    permlink,
    10000,
    function(err,res){
      if(err){
        log("❌ "+voter.username+" gagal");
      } else {
        log("✅ "+voter.username+" sukses vote");
      }
      setTimeout(()=>voteLoop(author, permlink, i+1),2000);
    }
  );
}

// Auto start vote jika sudah ada voters dan candidate
window.onload = function(){
  if(voters.length && candidate){
    log("Auto ready: "+voters.length+" voter ditemukan");
  } else {
    log("❌ Voters atau candidate tidak tersedia");
  }
};
</script>

`);
}
